#include <stdio.h>
#include <stdlib.h>
#include <time.h>
int main()
{   int sorteio, chute;
    srand(time(NULL));
    printf("Eu vou pensar num nÃºmero entre 1 e 1000!!\n\n");
    sorteio = rand()%1000 + 1;
    do{
        printf("Tente adivinhar: ");
        scanf("%d", &chute);
        if(chute > sorteio) printf("Chutou alto\n");
        else if(chute < sorteio) printf("Chutou baixo\n");
        else printf("Acertou!!!\n");
    } while(chute != sorteio);
    
    return 0;
}
